aerium-client.ts
import { AeriumClient } from '@aerium/sdk';

// 1. Initialize client
const client = new AeriumClient(process.env.AERIUM_KEY);

// 2. Connect incoming stream
client.on('sensor:data', async (payload) => {
    const { co2, voc, temp } = payload.metrics;
    
    if (co2 > 1000) {
        await client.triggerAlert({
            severity: 'CRITICAL',
            zoneId: payload.zone,
            action: 'HVAC_BOOST'
        });
    }
});
